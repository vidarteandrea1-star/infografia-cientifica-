                        var progreso = Math.min(elapsed / duracion, 1);

                        // Easing suave

                        var eased = 1 - Math.pow(1 - progreso, 3);

                        actual = Math.round(eased * objetivo);

                        fill.style.width = actual + '%';

                        valueLabel.textContent = actual + '%';

                        if (progreso < 1) {

                            requestAnimationFrame(animar);

                        }

                    }

                    requestAnimationFrame(animar);
